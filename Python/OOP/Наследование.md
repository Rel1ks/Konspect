# Наследование (Inheritance)

**Наследование** — механизм, при котором класс-потомок получает атрибуты и методы класса-родителя.

## 1. Базовое наследование

```python
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "..."

    def move(self):
        return f"{self.name} двигается"

class Dog(Animal):  # Dog наследует Animal
    def speak(self):
        return "Гав!"

class Cat(Animal):
    def speak(self):
        return "Мяу!"

dog = Dog("Бобик")
cat = Cat("Мурка")

print(dog.speak())    # Гав!
print(cat.speak())    # Мяу!
print(dog.move())     # Бобик двигается (унаследовано)
```

## 2. Переопределение методов (override)

Дочерний класс может полностью заменить метод родителя.

```python
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        return "..."

class Dog(Animal):
    def __init__(self, name, breed):
        self.name = name
        self.breed = breed

    def speak(self):
        return "Гав!"

    def wag_tail(self):
        return f"{self.name} виляет хвостом"
```

## 3. `super()` — вызов родительских методов

```python
class Animal:
    def __init__(self, name):
        self.name = name
        self.legs = 4

    def info(self):
        return f"{self.name}, {self.legs} лапы"

class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name)  # вызов __init__ родителя
        self.breed = breed

    def info(self):
        return f"{super().info()}, порода: {self.breed}"

dog = Dog("Бобик", "Такса")
print(dog.info())  # Бобик, 4 лапы, порода: Такса
```

## 4. Множественное наследование

Класс может наследовать от нескольких родителей.

```python
class Flyer:
    def fly(self):
        return "Летит"

    def move(self):
        return "Летит в небе"

class Swimmer:
    def swim(self):
        return "Плывёт"

    def move(self):
        return "Плывёт в воде"

class Duck(Flyer, Swimmer):
    pass

duck = Duck()
print(duck.fly())    # Летит
print(duck.swim())   # Плывёт
print(duck.move())   # Летит в небе (первый родитель — Flyer)
```

### MRO — Method Resolution Order

Порядок поиска методов при множественном наследовании.

```python
class A:
    def method(self):
        return "A"

class B(A):
    def method(self):
        return "B"

class C(A):
    def method(self):
        return "C"

class D(B, C):
    pass

d = D()
print(d.method())  # B (линеаризация C3)

print(D.__mro__)
# (<class 'D'>, <class 'B'>, <class 'C'>, <class 'A'>, <class 'object'>)
```

## 5. Diamond problem (ромбовидное наследование)

Python решает через MRO и `super()` с кооперативным наследованием.

```python
class A:
    def __init__(self):
        print("A.__init__")

class B(A):
    def __init__(self):
        print("B.__init__")
        super().__init__()

class C(A):
    def __init__(self):
        print("C.__init__")
        super().__init__()

class D(B, C):
    def __init__(self):
        print("D.__init__")
        super().__init__()

d = D()
# D.__init__
# B.__init__
# C.__init__
# A.__init__
# A вызывается один раз благодаря MRO
```

## 6. `isinstance()` и `issubclass()`

```python
class Animal: pass
class Dog(Animal): pass
class Cat(Animal): pass

dog = Dog()

print(isinstance(dog, Dog))      # True
print(isinstance(dog, Animal))   # True (Dog — подкласс Animal)
print(isinstance(dog, Cat))      # False

print(issubclass(Dog, Animal))   # True
print(issubclass(Animal, Dog))   # False
print(issubclass(Dog, object))   # True (все классы наследуют object)
```

## 7. Абстрактные классы (ABC)

Нельзя создать экземпляр. Определяет интерфейс для потомков.

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

    @abstractmethod
    def perimeter(self):
        pass

    def description(self):
        return f"Площадь: {self.area()}, Периметр: {self.perimeter()}"

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        return 3.14159 * self.radius ** 2

    def perimeter(self):
        return 2 * 3.14159 * self.radius

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height

    def area(self):
        return self.width * self.height

    def perimeter(self):
        return 2 * (self.width + self.height)

# s = Shape()  # TypeError
c = Circle(5)
r = Rectangle(4, 6)
print(c.description())
print(r.description())
```

### Абстрактные свойства

class Student(Person):

## 8. `@property` и наследование

```python
class Animal:
    def __init__(self, name):
        self._name = name

    @property
    def name(self):
        return self._name.upper()

    @name.setter
    def name(self, value):
        self._name = value

class Dog(Animal):
    @property
    def name(self):
        return f"Пёс {self._name}"

    @name.setter
    def name(self, value):
        self._name = f"{value} (собака)"

dog = Dog("Бобик")
print(dog.name)     # Пёс Бобик
dog.name = "Рекс"
print(dog.name)     # Пёст Рекс (собака)
```

## 9. Переопределение classmethod и staticmethod

```python
class Animal:
    @classmethod
    def create(cls, name):
        return cls(name)

    @staticmethod
    def description():
        return "Живое существо"

class Dog(Animal):
    @staticmethod
    def description():
        return "Четвероногий друг"

dog = Dog.create("Бобик")
print(isinstance(dog, Dog))  # True
print(Dog.description())     # Четвероногий друг
```

## 10. Приватные атрибуты и наследование

Приватные атрибуты (`__name`) не наследуются напрямую из-за name mangling.

```python
class Parent:
    def __init__(self):
        self.__secret = "родительский секрет"

    def get_secret(self):
        return self.__secret

class Child(Parent):
    def __init__(self):
        super().__init__()
        self.__secret = "детский секрет"

    def get_child_secret(self):
        return self.__secret

c = Child()
print(c.get_secret())        # родительский секрет (разные __secret)
print(c.get_child_secret())  # детский секрет
print(c._Parent__secret)     # родительский секрет
print(c._Child__secret)      # детский секрет
```

## 11. Наследование от встроенных типов

```python
class MyList(list):
    def sum(self):
        return sum(self)

    def average(self):
        return sum(self) / len(self) if self else 0

    def prepend(self, item):
        self.insert(0, item)

ml = MyList([1, 2, 3, 4])
ml.append(5)
ml.prepend(0)
print(ml)              # [0, 1, 2, 3, 4, 5]
print(ml.sum())        # 15
print(ml.average())    # 2.5
```

```python
class UpperDict(dict):
    def __setitem__(self, key, value):
        super().__setitem__(key.upper(), value)

    def __getitem__(self, key):
        return super().__getitem__(key.upper())

ud = UpperDict()
ud["name"] = "Анна"
print(ud["NAME"])      # Анна
print(ud)              # {'NAME': 'Анна'}
```

## 12. Композиция (альтернатива наследованию)

«Имеет» вместо «является».

```python
class Engine:
    def start(self):
        return "Двигатель запущен"

    def stop(self):
        return "Двигатель остановлен"

class Wheels:
    def rotate(self):
        return "Колёса крутятся"

class Car:
    def __init__(self):
        self.engine = Engine()  # композиция
        self.wheels = Wheels()

    def drive(self):
        return f"{self.engine.start()}, {self.wheels.rotate()}"

car = Car()
print(car.drive())
```

## 13. mixin

Небольшие классы, добавляющие конкретную функциональность.

```python
class JsonMixin:
    def to_json(self):
        import json
        return json.dumps(self.__dict__)

class LogMixin:
    def log(self, message):
        print(f"[LOG] {self.__class__.__name__}: {message}")

class TimestampMixin:
    def created_at(self):
        import datetime
        return datetime.datetime.now()

class User(JsonMixin, LogMixin, TimestampMixin):
    def __init__(self, name):
        self.name = name

user = User("Анна")
print(user.to_json())
user.log("создан")
print(user.created_at())
```
