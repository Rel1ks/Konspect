# Методы и дандер-методы (Methods & Dunder)

## Методы

### Instance methods

Принимают `self` — работают с конкретным экземпляром.

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def greet(self):
        return f"Привет, {self.name}!"

Person.greet        # function
p = Person("Анна", 25)
p.greet             # bound method (self привязан)
```

### Class methods (@classmethod)

Принимают `cls` — работают с классом. Могут создавать экземпляры класса (factory methods).

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    @classmethod
    def from_birth_year(cls, name, year):
        return cls(name, 2026 - year)

    @classmethod
    def from_string(cls, data):
        name, age = data.split(",")
        return cls(name.strip(), int(age.strip()))

p1 = Person.from_birth_year("Анна", 2000)
p2 = Person.from_string("Борис, 30")
```

### Static methods (@staticmethod)

Не принимают ни `self`, ни `cls`. Обычная функция внутри класса для логической группировки.

```python
class StringUtils:
    @staticmethod
    def is_palindrome(s):
        return s == s[::-1]

    @staticmethod
    def reverse(s):
        return s[::-1]

print(StringUtils.is_palindrome("radar"))
print(StringUtils.reverse("hello"))
```

### Bound, Unbound методы

```python
class A:
    def method(self):
        return "метод"

a = A()
print(a.method)  # <bound method A.method of <...>>
print(A.method)  # <function A.method at ...>

# вызов через класс требует явного self
print(A.method(a))  # "метод"
```

## Дандер-методы (Dunder / Magic methods)

**Dunder** = **D**ouble **Under**score (`__method__`). Позволяют переопределять поведение объектов.

### Создание и удаление

```python
class Example:
    def __new__(cls, *args, **kwargs):
        """Создание объекта (редко переопределяют)"""
        print(f"__new__: создание {cls}")
        return super().__new__(cls)

    def __init__(self, value):
        """Инициализация объекта"""
        self.value = value
        print(f"__init__: {value}")

    def __del__(self):
        """Удаление объекта (не гарантируется вызов)"""
        print(f"__del__: {self.value}")

e = Example(42)
```

### Строковое представление

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __str__(self):
        """Для пользователя (print, str)"""
        return f"{self.name}, {self.age} лет"

    def __repr__(self):
        """Для разработчика (repr, отладка)"""
        return f"Person('{self.name}', {self.age})"

    def __format__(self, format_spec):
        """Форматирование через f-string / format()"""
        if format_spec == "name":
            return self.name
        return str(self)

p = Person("Анна", 25)
print(str(p))          # Анна, 25 лет
print(repr(p))         # Person('Анна', 25)
print(f"{p:name}")     # Анна
```

### Сравнение

| Дандер | Оператор |
|---|---|
| `__eq__` | `==` |
| `__ne__` | `!=` |
| `__lt__` | `<` |
| `__le__` | `<=` |
| `__gt__` | `>` |
| `__ge__` | `>=` |
| `__hash__` | `hash()` (требуется для set/dict) |

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __eq__(self, other):
        if not isinstance(other, Person):
            return NotImplemented
        return self.name == other.name and self.age == other.age

    def __lt__(self, other):
        return self.age < other.age

    def __le__(self, other):
        return self.age <= other.age

    def __hash__(self):
        return hash((self.name, self.age))

p1 = Person("Анна", 25)
p2 = Person("Анна", 25)
p3 = Person("Борис", 30)

print(p1 == p2)   # True
print(p1 < p3)    # True
print(p1 <= p3)   # True
print(p1 != p3)   # True

# можно в set
s = {p1, p3}
# можно в dict
d = {p1: "объект"}
```

### Обратные операции

Если `a + b` не работает, Python пробует `b.__radd__(a)`.

| Дандер | Оператор |
|---|---|
| `__radd__` | `+` (другой объект слева) |
| `__rsub__` | `-` |
| `__rmul__` | `*` |
| `__rtruediv__` | `/` |

### In-place операции

| Дандер | Оператор |
|---|---|
| `__iadd__` | `+=` |
| `__isub__` | `-=` |
| `__imul__` | `*=` |

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)

    def __sub__(self, other):
        return Vector(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)

    def __rmul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)

    def __neg__(self):
        return Vector(-self.x, -self.y)

    def __iadd__(self, other):
        self.x += other.x
        self.y += other.y
        return self

    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

v1 = Vector(3, 4)
v2 = Vector(1, 2)

print(v1 + v2)    # Vector(4, 6)
print(v1 - v2)    # Vector(2, 2)
print(v1 * 3)     # Vector(9, 12)
print(3 * v1)     # Vector(9, 12) (__rmul__)
print(-v1)        # Vector(-3, -4)

v1 += v2
print(v1)         # Vector(4, 6)
```

### Контейнеры и последовательности

| Дандер | Описание |
|---|---|
| `__len__` | `len(obj)` |
| `__getitem__` | `obj[key]` |
| `__setitem__` | `obj[key] = val` |
| `__delitem__` | `del obj[key]` |
| `__contains__` | `x in obj` |
| `__iter__` | `iter(obj)` |
| `__next__` | `next(obj)` |
| `__reversed__` | `reversed(obj)` |
| `__missing__` | Доступ к отсутствующему ключу (для dict) |

```python
class Range:
    def __init__(self, start, end):
        self.start = start
        self.end = end

    def __len__(self):
        return max(0, self.end - self.start)

    def __getitem__(self, index):
        if index < 0 or index >= len(self):
            raise IndexError
        return self.start + index

    def __contains__(self, item):
        return self.start <= item < self.end

    def __iter__(self):
        return iter(range(self.start, self.end))

    def __reversed__(self):
        return range(self.end - 1, self.start - 1, -1)

r = Range(3, 8)
print(len(r))         # 5
print(r[2])           # 5
print(5 in r)         # True
print(list(r))        # [3, 4, 5, 6, 7]
```

### Преобразование типов

| Дандер | Описание |
|---|---|
| `__int__` | `int(obj)` |
| `__float__` | `float(obj)` |
| `__bool__` | `bool(obj)` |
| `__str__` | `str(obj)` |
| `__bytes__` | `bytes(obj)` |
| `__complex__` | `complex(obj)` |

```python
class Score:
    def __init__(self, value):
        self.value = value

    def __int__(self):
        return self.value

    def __float__(self):
        return float(self.value)

    def __bool__(self):
        return self.value != 0

    def __str__(self):
        grade = "A" if self.value >= 90 else "B" if self.value >= 75 else "C"
        return f"{self.value} ({grade})"

s = Score(85)
print(int(s))    # 85
print(float(s))  # 85.0
print(bool(Score(0)))  # False
print(str(s))    # 85 (B)
```

### Вызов объекта

```python
class Multiplier:
    def __init__(self, factor):
        self.factor = factor

    def __call__(self, x):
        return x * self.factor

double = Multiplier(2)
triple = Multiplier(3)

print(double(5))   # 10
print(triple(5))   # 15
```

### Контекстный менеджер (`with`)

```python
class ManagedFile:
    def __init__(self, filename):
        self.filename = filename

    def __enter__(self):
        print(f"Открытие {self.filename}")
        self.file = open(self.filename, "w")
        return self.file

    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
        print(f"Закрытие {self.filename}")
        return False  # не подавлять исключения

with ManagedFile("test.txt") as f:
    f.write("Hello")
```

### Дескрипторы

```python
class PositiveNumber:
    def __get__(self, obj, objtype=None):
        return obj.__dict__.get(self.name, 0)

    def __set__(self, obj, value):
        if value < 0:
            raise ValueError("Только положительные числа")
        obj.__dict__[self.name] = value

    def __set_name__(self, owner, name):
        self.name = f"_{name}"

class Order:
    price = PositiveNumber()
    quantity = PositiveNumber()

    def __init__(self, price, quantity):
        self.price = price
        self.quantity = quantity

order = Order(100, 2)
print(order.price)
# order.price = -50  # ValueError
```
