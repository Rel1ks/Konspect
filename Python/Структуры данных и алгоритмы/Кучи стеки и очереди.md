# Кучи, Стеки и Очереди (Heaps, Stacks & Queues)

## Стек (Stack)

**Стек** — структура данных, работающая по принципу **LIFO** (Last In, First Out — последним пришёл, первым вышел).

### Операции

| Операция | Описание | Сложность |
|---|---|---|
| `push(item)` | Добавить на вершину | O(1) |
| `pop()` | Удалить и вернуть с вершины | O(1) |
| `peek()` / `top()` | Посмотреть вершину без удаления | O(1) |
| `is_empty()` | Проверка на пустоту | O(1) |
| `size()` | Количество элементов | O(1) |

### Реализация через список

```python
class Stack:
    def __init__(self):
        self.items = []

    def push(self, item):
        self.items.append(item)

    def pop(self):
        if self.is_empty():
            return None
        return self.items.pop()

    def peek(self):
        if self.is_empty():
            return None
        return self.items[-1]

    def is_empty(self):
        return len(self.items) == 0

    def size(self):
        return len(self.items)

    def __repr__(self):
        return f"Stack({self.items})"

# использование
s = Stack()
s.push(1)
s.push(2)
s.push(3)
print(s.pop())     # 3
print(s.peek())    # 2
print(s.size())    # 2
```

### Примеры использования

```python
# проверка скобок
def is_balanced(text):
    stack = []
    pairs = {")": "(", "]": "[", "}": "{"}

    for ch in text:
        if ch in "([{":
            stack.append(ch)
        elif ch in ")]}":
            if not stack or stack.pop() != pairs[ch]:
                return False
    return len(stack) == 0

print(is_balanced("({[]})"))  # True
print(is_balanced("({[})"))   # False

# обратная польская нотация
def eval_rpn(tokens):
    stack = []
    for token in tokens:
        if token in "+-*/":
            b = stack.pop()
            a = stack.pop()
            if token == "+": stack.append(a + b)
            elif token == "-": stack.append(a - b)
            elif token == "*": stack.append(a * b)
            elif token == "/": stack.append(int(a / b))
        else:
            stack.append(int(token))
    return stack.pop()

print(eval_rpn(["2", "1", "+", "3", "*"]))  # 9
```

## Очередь (Queue)

**Очередь** — структура данных, работающая по принципу **FIFO** (First In, First Out — первым пришёл, первым вышел).

### Операции

| Операция | Описание | Сложность |
|---|---|---|
| `enqueue(item)` | Добавить в конец | O(1) |
| `dequeue()` | Удалить и вернуть из начала | O(1) |
| `front()` | Посмотреть первый элемент | O(1) |
| `is_empty()` | Проверка на пустоту | O(1) |
| `size()` | Количество элементов | O(1) |

### Реализация через collections.deque

```python
from collections import deque

class Queue:
    def __init__(self):
        self.items = deque()

    def enqueue(self, item):
        self.items.append(item)

    def dequeue(self):
        if self.is_empty():
            return None
        return self.items.popleft()

    def front(self):
        if self.is_empty():
            return None
        return self.items[0]

    def is_empty(self):
        return len(self.items) == 0

    def size(self):
        return len(self.items)

    def __repr__(self):
        return f"Queue({list(self.items)})"

# использование
q = Queue()
q.enqueue(1)
q.enqueue(2)
q.enqueue(3)
print(q.dequeue())  # 1
print(q.front())    # 2
```

### Реализация через список (менее эффективно)

```python
# плохо — pop(0) это O(n)
class BadQueue:
    def __init__(self):
        self.items = []

    def enqueue(self, item):
        self.items.append(item)

    def dequeue(self):
        return self.items.pop(0)  # O(n)
```

### Примеры использования

```python
# печать задач в порядке поступления
tasks = Queue()
tasks.enqueue("Задача 1")
tasks.enqueue("Задача 2")
tasks.enqueue("Задача 3")

while not tasks.is_empty():
    print(f"Выполняется: {tasks.dequeue()}")

# очереди — основа BFS (поиск в ширину)
def bfs(graph, start):
    visited = {start}
    queue = Queue()
    queue.enqueue(start)

    while not queue.is_empty():
        node = queue.dequeue()
        print(node, end=" ")
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.enqueue(neighbor)

graph = {
    1: [2, 3],
    2: [4, 5],
    3: [6],
}
# bfs(graph, 1)  # 1 2 3 4 5 6
```

### Двусторонняя очередь (Deque)

```python
from collections import deque

d = deque()

d.append(1)        # добавить справа
d.appendleft(2)    # добавить слева
d.pop()            # удалить справа
d.popleft()        # удалить слева

d.extend([3, 4])        # добавить несколько справа
d.extendleft([5, 6])    # добавить несколько слева
d.rotate(1)             # циклический сдвиг вправо
d.rotate(-1)            # циклический сдвиг влево
```

## Очередь с приоритетом (Priority Queue)

**Очередь с приоритетом** — элементы извлекаются по приоритету (не по порядку добавления).

```python
import heapq

class PriorityQueue:
    def __init__(self):
        self.items = []
        self.index = 0

    def push(self, item, priority):
        heapq.heappush(self.items, (priority, self.index, item))
        self.index += 1

    def pop(self):
        if self.is_empty():
            return None
        return heapq.heappop(self.items)[-1]

    def is_empty(self):
        return len(self.items) == 0

    def __len__(self):
        return len(self.items)

pq = PriorityQueue()
pq.push("низкий", 3)
pq.push("высокий", 1)
pq.push("средний", 2)

print(pq.pop())  # высокий
print(pq.pop())  # средний
print(pq.pop())  # низкий
```

## Куча (Heap)

**Куча** — бинарное дерево, где каждый родитель меньше (min-heap) или больше (max-heap) своих потомков.

### Min-куча

```python
import heapq

# heapq в Python — min-heap
heap = []
heapq.heappush(heap, 5)
heapq.heappush(heap, 1)
heapq.heappush(heap, 3)
heapq.heappush(heap, 7)
heapq.heappush(heap, 2)

print(heap)           # [1, 2, 3, 7, 5]
print(heapq.heappop(heap))   # 1
print(heapq.heappop(heap))   # 2
print(heapq.heappop(heap))   # 3
```

### Max-куча (через отрицательные числа)

```python
heap = []
heapq.heappush(heap, -5)
heapq.heappush(heap, -1)
heapq.heappush(heap, -3)
heapq.heappush(heap, -7)
heapq.heappush(heap, -2)

print(-heapq.heappop(heap))  # 7
print(-heapq.heappop(heap))  # 5
```

### Основные операции heapq

| Функция | Описание | Сложность |
|---|---|---|
| `heappush(heap, item)` | Добавить элемент | O(log n) |
| `heappop(heap)` | Извлечь минимальный | O(log n) |
| `heapify(list)` | Преобразовать список в кучу | O(n) |
| `heapreplace(heap, item)` | Pop + push | O(log n) |
| `heappushpop(heap, item)` | Push + pop | O(log n) |
| `nlargest(n, iter)` | n наибольших | O(n log n) |
| `nsmallest(n, iter)` | n наименьших | O(n log n) |

```python
# heapify — превратить список в кучу
nums = [5, 3, 8, 1, 2, 7]
heapq.heapify(nums)
print(nums)  # [1, 2, 7, 3, 5, 8]

# nlargest / nsmallest
print(heapq.nlargest(3, [3, 1, 4, 1, 5, 9, 2]))  # [9, 5, 4]
print(heapq.nsmallest(3, [3, 1, 4, 1, 5, 9, 2]))  # [1, 1, 2]
```

### Примеры использования

```python
# найти k наибольших элементов
def top_k(nums, k):
    heap = nums[:k]
    heapq.heapify(heap)
    for num in nums[k:]:
        if num > heap[0]:
            heapq.heapreplace(heap, num)
    return sorted(heap, reverse=True)

print(top_k([3, 1, 5, 7, 2, 8, 4], 3))  # [8, 7, 5]

# объединение k отсортированных списков
def merge_k_lists(lists):
    heap = []
    for i, lst in enumerate(lists):
        if lst:
            heapq.heappush(heap, (lst[0], i, 0))

    result = []
    while heap:
        val, list_idx, elem_idx = heapq.heappop(heap)
        result.append(val)
        if elem_idx + 1 < len(lists[list_idx]):
            next_val = lists[list_idx][elem_idx + 1]
            heapq.heappush(heap, (next_val, list_idx, elem_idx + 1))
    return result

print(merge_k_lists([[1, 4, 7], [2, 5, 8], [3, 6, 9]]))
```

## Сравнение

| Структура | Принцип | Вставка | Извлечение | Доступ |
|---|---|---|---|---|
| **Стек** | LIFO | O(1) | O(1) (с конца) | Только вершина |
| **Очередь** | FIFO | O(1) | O(1) (с начала) | Только начало |
| **Deque** | Оба конца | O(1) | O(1) | Оба конца |
| **Priority Queue** | По приоритету | O(log n) | O(log n) | Только min/max |
| **Куча (heap)** | Min/Max | O(log n) | O(log n) | Только min/max |

### Когда что использовать

```python
# стек — отмена действий
class UndoManager:
    def __init__(self):
        self.history = []

    def do(self, action):
        self.history.append(action)

    def undo(self):
        return self.history.pop() if self.history else None

# очередь — задачи по порядку
from collections import deque

class TaskScheduler:
    def __init__(self):
        self.tasks = deque()

    def add_task(self, task):
        self.tasks.append(task)

    def next_task(self):
        return self.tasks.popleft() if self.tasks else None

# куча — найти топ-N
def top_n_scores(scores, n):
    return heapq.nlargest(n, scores)

# очередь с приоритетом — срочные задачи
def process_tasks(tasks):
    heap = []
    for task, priority in tasks:
        heapq.heappush(heap, (priority, task))

    while heap:
        priority, task = heapq.heappop(heap)
        print(f"[{priority}] {task}")
```
