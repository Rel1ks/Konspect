# Бинарные деревья поиска (Binary Search Trees)

**Бинарное дерево поиска (BST)** — дерево, где для каждого узла:

- Все значения в **левом** поддереве **меньше** значения узла
- Все значения в **правом** поддереве **больше** значения узла
- Левое и правое поддеревья тоже являются BST

## 1. Узел дерева

```python
class TreeNode:
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None
```

## 2. Базовые операции

### Вставка

```python
def insert(root, value):
    if root is None:
        return TreeNode(value)
    if value < root.value:
        root.left = insert(root.left, value)
    else:
        root.right = insert(root.right, value)
    return root

# использование
root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)
```

### Поиск

```python
def search(root, value):
    if root is None or root.value == value:
        return root
    if value < root.value:
        return search(root.left, value)
    else:
        return search(root.right, value)

# используется редко, обычно используют in
```

### Удаление

```python
def min_value_node(node):
    current = node
    while current.left:
        current = current.left
    return current

def delete(root, value):
    if root is None:
        return root

    if value < root.value:
        root.left = delete(root.left, value)
    elif value > root.value:
        root.right = delete(root.right, value)
    else:
        # случай 1: нет детей
        if root.left is None and root.right is None:
            return None
        # случай 2: один ребёнок
        elif root.left is None:
            return root.right
        elif root.right is None:
            return root.left
        # случай 3: два ребёнка
        temp = min_value_node(root.right)
        root.value = temp.value
        root.right = delete(root.right, temp.value)
    return root
```

## 3. Обходы дерева (Tree Traversals)

```python
def inorder(root):
    """Левый → Корень → Правый (отсортированный порядок) O(n)"""
    if root:
        inorder(root.left)
        print(root.value, end=" ")
        inorder(root.right)

def preorder(root):
    """Корень → Левый → Правый O(n)"""
    if root:
        print(root.value, end=" ")
        preorder(root.left)
        preorder(root.right)

def postorder(root):
    """Левый → Правый → Корень O(n)"""
    if root:
        postorder(root.left)
        postorder(root.right)
        print(root.value, end=" ")

def level_order(root):
    """Обход по уровням (BFS) O(n)"""
    if not root:
        return
    from collections import deque
    queue = deque([root])
    while queue:
        node = queue.popleft()
        print(node.value, end=" ")
        if node.left:
            queue.append(node.left)
        if node.right:
            queue.append(node.right)

# demo
root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)

print("inorder:   ", end=""); inorder(root)     # 2 3 4 5 6 7 8
print("\npreorder:  ", end=""); preorder(root)    # 5 3 2 4 7 6 8
print("\npostorder: ", end=""); postorder(root)   # 2 4 3 6 8 7 5
print("\nlevel:     ", end=""); level_order(root) # 5 3 7 2 4 6 8
```

## 4. Полная реализация BST

```python
class BST:
    class _Node:
        def __init__(self, value):
            self.value = value
            self.left = None
            self.right = None

    def __init__(self):
        self.root = None
        self._size = 0

    def insert(self, value):
        self.root = self._insert(self.root, value)
        self._size += 1

    def _insert(self, node, value):
        if node is None:
            return self._Node(value)
        if value < node.value:
            node.left = self._insert(node.left, value)
        else:
            node.right = self._insert(node.right, value)
        return node

    def search(self, value):
        return self._search(self.root, value)

    def _search(self, node, value):
        if node is None:
            return False
        if value == node.value:
            return True
        if value < node.value:
            return self._search(node.left, value)
        return self._search(node.right, value)

    def delete(self, value):
        self.root = self._delete(self.root, value)
        self._size -= 1

    def _delete(self, node, value):
        if node is None:
            return None
        if value < node.value:
            node.left = self._delete(node.left, value)
        elif value > node.value:
            node.right = self._delete(node.right, value)
        else:
            if node.left is None:
                return node.right
            if node.right is None:
                return node.left
            min_node = self._min(node.right)
            node.value = min_node.value
            node.right = self._delete(node.right, min_node.value)
        return node

    def _min(self, node):
        while node.left:
            node = node.left
        return node

    def min(self):
        if not self.root:
            return None
        return self._min(self.root).value

    def max(self):
        node = self.root
        while node and node.right:
            node = node.right
        return node.value if node else None

    def height(self):
        return self._height(self.root)

    def _height(self, node):
        if node is None:
            return -1
        return 1 + max(self._height(node.left), self._height(node.right))

    def inorder(self):
        result = []
        self._inorder(self.root, result)
        return result

    def _inorder(self, node, result):
        if node:
            self._inorder(node.left, result)
            result.append(node.value)
            self._inorder(node.right, result)

    def __contains__(self, value):
        return self.search(value)

    def __len__(self):
        return self._size

    def __repr__(self):
        return str(self.inorder())

# использование
bst = BST()
for v in [5, 3, 7, 2, 4, 6, 8]:
    bst.insert(v)

print(bst)            # [2, 3, 4, 5, 6, 7, 8]
print(4 in bst)       # True
print(9 in bst)       # False
print(bst.min())      # 2
print(bst.max())      # 8
print(bst.height())   # 2

bst.delete(3)
print(bst)            # [2, 4, 5, 6, 7, 8]
```

## 5. Сложность операций

| Операция | Средняя (сбалансированное) | Худшая (вырожденное) |
|---|---|---|
| Вставка | O(log n) | O(n) |
| Поиск | O(log n) | O(n) |
| Удаление | O(log n) | O(n) |
| Обход | O(n) | O(n) |
| Min/Max | O(log n) | O(n) |

## 6. Сбалансированные деревья

Когда BST вырождается в связный список (если вставлять отсортированные данные), все операции становятся O(n).

```python
# вырожденное дерево (как список)
root = None
for v in [1, 2, 3, 4, 5]:  # отсортированные данные
    root = insert(root, v)
# дерево превращается в цепочку вправо
```

### Самобалансирующиеся деревья

| Тип | Описание |
|---|---|
| **AVL-дерево** | Высота поддеревьев отличается не более чем на 1 |
| **Красно-чёрное дерево** | Цвета узлов, балансировка через повороты |
| **B-дерево** | Много ключей в узле, используется в БД |

В Python `set` и `dict` — хеш-таблицы, не деревья. Но `sortedcontainers` использует дерево.

## 7. Практические примеры

### Проверка, является ли дерево BST

```python
def is_bst(node, min_val=float("-inf"), max_val=float("inf")):
    if node is None:
        return True
    if not min_val < node.value < max_val:
        return False
    return (is_bst(node.left, min_val, node.value) and
            is_bst(node.right, node.value, max_val))

root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)
print(is_bst(root))  # True
```

### Найти k-й наименьший элемент

```python
def kth_smallest(root, k):
    stack = []
    current = root
    count = 0

    while stack or current:
        while current:
            stack.append(current)
            current = current.left
        current = stack.pop()
        count += 1
        if count == k:
            return current.value
        current = current.right
    return None

root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)
print(kth_smallest(root, 3))  # 4
```

### Поиск наименьшего общего предка (LCA)

```python
def lca(root, a, b):
    if root is None:
        return None
    if a < root.value and b < root.value:
        return lca(root.left, a, b)
    if a > root.value and b > root.value:
        return lca(root.right, a, b)
    return root.value

root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)
print(lca(root, 2, 4))  # 3
print(lca(root, 2, 8))  # 5
```

### Обход в глубину (DFS) — итеративно

```python
def dfs_inorder(root):
    stack, result = [], []
    current = root
    while stack or current:
        while current:
            stack.append(current)
            current = current.left
        current = stack.pop()
        result.append(current.value)
        current = current.right
    return result

def dfs_preorder(root):
    if not root:
        return []
    stack, result = [root], []
    while stack:
        node = stack.pop()
        result.append(node.value)
        if node.right:
            stack.append(node.right)
        if node.left:
            stack.append(node.left)
    return result

def dfs_postorder(root):
    if not root:
        return []
    stack, result = [(root, False)], []
    while stack:
        node, visited = stack.pop()
        if visited:
            result.append(node.value)
        else:
            stack.append((node, True))
            if node.right:
                stack.append((node.right, False))
            if node.left:
                stack.append((node.left, False))
    return result
```

### Сериализация / Десериализация

```python
def serialize(root):
    """Преобразовать дерево в строку (preorder)"""
    if not root:
        return "null"
    return f"{root.value},{serialize(root.left)},{serialize(root.right)}"

def deserialize(data):
    """Восстановить дерево из строки"""
    def helper(values):
        val = next(values)
        if val == "null":
            return None
        node = TreeNode(int(val))
        node.left = helper(values)
        node.right = helper(values)
        return node

    return helper(iter(data.split(",")))

root = None
for v in [5, 3, 7, 2, 4, 6, 8]:
    root = insert(root, v)

s = serialize(root)
print(s)  # "5,3,2,null,null,4,null,null,7,6,null,null,8,null,null"
restored = deserialize(s)
```
